macro(add_source_group FILTER_NAME SOURCE_PATH TARGET_LIST)
    file(GLOB TEMP_SRC
        ${SOURCE_PATH}/*.h
        ${SOURCE_PATH}/*.cpp
        ${SOURCE_PATH}/*.cc
        ${SOURCE_PATH}/*.c
    )
    source_group(${FILTER_NAME} FILES ${TEMP_SRC})
    list(APPEND ${TARGET_LIST} ${TEMP_SRC})
endmacro(add_source_group)
